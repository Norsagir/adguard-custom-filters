!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
!!! PC WEB BROWSER EXTENSION !!!
!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!

!! Home
! Shelf section
youtube.com##[page-subtype="home"] ytd-rich-section-renderer:has([is-shorts])

!! Channels
! Shelf section
youtube.com##[page-subtype="channels"] ytd-reel-shelf-renderer:has(.ytd-reel-shelf-renderer:has-text(/(^| )Shorts( |$)/i))

!! Subscriptions
! Shelf section
youtube.com##[page-subtype="subscriptions"] ytd-rich-section-renderer:has([is-shorts])

!! Search
! Shelf sections
youtube.com##[page-subtype="search"] grid-shelf-view-model:has(yt-section-header-view-model:has(yt-shelf-header-layout:has-text(Shorts)))


!!!!! DISABLED because no personal use, but can be reactivated !!!!!
!! Global
! Remove Shorts button in the sidebar menu
!youtube.com##ytd-guide-entry-renderer:has(yt-formatted-string:has-text(/^Shorts$/i))
! Remove Shorts button if the sidebar menu is mini
!youtube.com##ytd-mini-guide-entry-renderer:has([title="Shorts"])

!! Channels
! Remove button
!youtube.com##[page-subtype="channels"] yt-tab-shape:has-text(/^Shorts$/)

!! Search
! Remove button
!youtube.com##[page-subtype="search"] yt-chip-cloud-chip-renderer:has-text(Shorts)

!! History
! Remove every video with the tag 'Shorts' in overlay
!youtube.com##[page-subtype="history"] ytd-video-renderer:has([overlay-style="SHORTS"])
! Remove button
!youtube.com##[page-subtype="history"] chip-view-model:has-text(Shorts)
